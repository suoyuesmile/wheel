# 实现一个 apply

# 完整代码

Function.prototype.__apply = function() {
    var arg = Array.prototype.shift.call(arguments);
    var params = Array.prototype.shift.call(arguments);
    arg.fn = this;
    var applyFn = arg.fn(...params);
    delete arg.fn;
    return applyFn;
}
