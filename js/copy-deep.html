function __copy_deep(origin) {
    const newObj = {};
    for (const key in origin) {
        newObj[key] = origin[key];
        if (typeof origin[key] === 'object') {
            return __copy_deep(origin[key])
        } else {
            return origin[key];
        }
    }
}
